**Step 1 — Create `spd-ready/src/lib/dal/auth.ts`:**

```typescript
import { cache } from 'react'
import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'

/**
 * Get the currently authenticated user.
 * Uses getUser() — contacts the Supabase auth server to verify the JWT.
 * Never uses getSession() — that reads an unverified local JWT.
 *
 * Memoized with React cache() so multiple calls in one render cycle
 * do not trigger duplicate auth server round-trips.
 */
export const getCurrentUser = cache(async () => {
  const supabase = await createClient()
  const {
    data: { user },
    error,
  } = await supabase.auth.getUser()

  if (error || !user) {
    return null
  }
  return user
})

/**
 * Get the role of the current user.
 * Reads from app_metadata.role (server-controlled, not user-editable).
 * Returns null if not authenticated.
 */
export const getRole = cache(async (): Promise<'student' | 'hospital' | 'admin' | null> => {
  const user = await getCurrentUser()
  if (!user) return null
  return (user.app_metadata?.role as 'student' | 'hospital' | 'admin') ?? null
})

/**
 * Require a specific role. Redirects if not authenticated or wrong role.
 * Use at the top of Server Actions and Route Handlers that need role enforcement.
 *
 * @param role - The required role
 * @param redirectTo - Where to redirect on failure (default: '/unauthorized')
 */
export async function requireRole(
  role: 'student' | 'hospital' | 'admin',
  redirectTo = '/unauthorized'
): Promise<NonNullable<Awaited<ReturnType<typeof getCurrentUser>>>> {
  const user = await getCurrentUser()
  if (!user) redirect('/login')

  const userRole = user.app_metadata?.role
  if (userRole !== role) redirect(redirectTo)

  return user
}
```

**Step 2 — Create `spd-ready/src/actions/auth.ts`:**

```typescript
'use server'

import { revalidatePath } from 'next/cache'
import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'
import { createClient as createAdminClient } from '@supabase/supabase-js'

// ============================================================
// Sign In
// ============================================================
export async function signInAction(formData: FormData) {
  const email = formData.get('email') as string
  const password = formData.get('password') as string

  if (!email || !password) {
    return { error: 'Email and password are required.' }
  }

  const supabase = await createClient()
  const { error } = await supabase.auth.signInWithPassword({ email, password })

  if (error) {
    return { error: error.message }
  }

  revalidatePath('/', 'layout')
  redirect('/dashboard')
}

// ============================================================
// Sign Up
// Role is stored in app_metadata (server-controlled) via Admin API.
// Users cannot modify app_metadata — only service_role key can.
// A row is also inserted into public.users for DAL queries and RLS.
// ============================================================
export async function signUpAction(formData: FormData) {
  const email = formData.get('email') as string
  const password = formData.get('password') as string
  const role = formData.get('role') as 'student' | 'hospital'

  if (!email || !password || !role) {
    return { error: 'Email, password, and role are required.' }
  }

  if (!['student', 'hospital'].includes(role)) {
    return { error: 'Role must be student or hospital.' }
  }

  // Step 1: Create auth user (basic signup — no role in user_metadata)
  const supabase = await createClient()
  const { data: signUpData, error: signUpError } = await supabase.auth.signUp({
    email,
    password,
    options: {
      emailRedirectTo: `${process.env.NEXT_PUBLIC_SUPABASE_URL}/auth/v1/callback`,
    },
  })

  if (signUpError) {
    return { error: signUpError.message }
  }

  const userId = signUpData.user?.id
  if (!userId) {
    return { error: 'User creation failed — no user ID returned.' }
  }

  // Step 2: Set role in app_metadata using service_role Admin API.
  // CRITICAL: Only the service_role key can write to app_metadata.
  // This prevents users from elevating their own role.
  const supabaseAdmin = createAdminClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!
  )

  const { error: metaError } = await supabaseAdmin.auth.admin.updateUserById(userId, {
    app_metadata: { role },
  })

  if (metaError) {
    return { error: `Failed to assign role: ${metaError.message}` }
  }

  // Step 3: Insert a row into public.users.
  // This is the authoritative source for the custom_access_token_hook.
  // Use admin client so RLS does not block the insert.
  const { error: userRowError } = await supabaseAdmin
    .from('users')
    .insert({ id: userId, email, role })

  if (userRowError) {
    return { error: `Failed to create user record: ${userRowError.message}` }
  }

  // Sign in immediately after signup (if email confirmation is disabled in Supabase settings)
  // For demo/development, email confirmation can be disabled in Supabase Dashboard
  revalidatePath('/', 'layout')
  redirect('/dashboard')
}

// ============================================================
// Sign Out
// ============================================================
export async function signOutAction() {
  const supabase = await createClient()
  await supabase.auth.signOut()
  revalidatePath('/', 'layout')
  redirect('/login')
}

// ============================================================
// Reset Password
// Sends a password reset email via Supabase Auth.
// ============================================================
export async function resetPasswordAction(formData: FormData) {
  const email = formData.get('email') as string

  if (!email) {
    return { error: 'Email is required.' }
  }

  const supabase = await createClient()
  const { error } = await supabase.auth.resetPasswordForEmail(email, {
    redirectTo: `${process.env.NEXT_PUBLIC_SITE_URL ?? 'http://localhost:3000'}/auth/callback?next=/reset-password`,
  })

  if (error) {
    return { error: error.message }
  }

  return { success: 'Check your email for a password reset link.' }
}
```

**Step 3 — Create the auth callback Route Handler at `spd-ready/src/app/auth/callback/route.ts`:**

```typescript
import { NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'

/**
 * Handles email confirmation and password reset callbacks.
 * Supabase Auth redirects here with a one-time `code` param.
 * Exchange the code for a session, then redirect to the intended destination.
 */
export async function GET(request: Request) {
  const { searchParams, origin } = new URL(request.url)
  const code = searchParams.get('code')
  const next = searchParams.get('next') ?? '/dashboard'

  if (code) {
    const supabase = await createClient()
    const { error } = await supabase.auth.exchangeCodeForSession(code)
    if (!error) {
      return NextResponse.redirect(`${origin}${next}`)
    }
  }

  // If no code or exchange failed, redirect to login with error indicator
  return NextResponse.redirect(`${origin}/login?error=auth_callback_failed`)
}
```

import { NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'

/**
 * Sign-out route used by the nav "Sign out" link.
 * A GET-based sign-out is acceptable here because the actual
 * session destruction is handled server-side via Supabase.
 */
export async function GET(request: Request) {
  const supabase = await createClient()
  await supabase.auth.signOut()

  const origin = new URL(request.url).origin
  return NextResponse.redirect(`${origin}/login`)
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser form → `signUpAction` (Server Action) | User-controlled email, password, role cross here |
| `signUpAction` → Supabase Admin API | Service role key used; must be server-only |
| `signInAction` → Supabase Auth | Credentials cross this boundary |
| `/auth/callback` Route Handler → Supabase | One-time code exchange; code is from email link |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | `signUpAction` — user could submit `role=admin` via form manipulation | mitigate | Server Action validates `role` is only `'student'` or `'hospital'`. Admin accounts created out-of-band (Supabase dashboard or separate admin script). Role stored in `app_metadata` via service key — not settable by user. |
| T-03-02 | Tampering | `SUPABASE_SERVICE_ROLE_KEY` exposed to client | mitigate | `createAdminClient` is called inside a Server Action (`'use server'`). Server Actions run server-side only. `SUPABASE_SERVICE_ROLE_KEY` has no `NEXT_PUBLIC_` prefix. |
| T-03-03 | Spoofing | `signInAction` brute force attack | accept | Supabase Auth has built-in rate limiting on sign-in attempts. Additional IP-level rate limiting at Vercel edge is out of scope for MVP. |
| T-03-04 | Tampering | CSRF on form submission | mitigate | Next.js Server Actions include automatic CSRF protection (Origin header check). No additional CSRF token needed. |
| T-03-05 | Spoofing | `auth/callback` code replay | mitigate | Supabase one-time codes are single-use and expire. `exchangeCodeForSession` fails if code is already used or expired. |
| T-03-06 | Information Disclosure | Error messages revealing whether an email is registered | accept | Supabase Auth returns generic errors for sign-in failures in production mode. "Check your email and password" message does not confirm whether the email exists. Acceptable for MVP. |
</threat_model>
